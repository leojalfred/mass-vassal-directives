# Leo VI's Mass Vassal Directives - the rule engine
#
# A rule set is an ordered list of priorities. Each priority is a small tree of
# nodes; the first priority that assigns a directive wins, and a vassal that no
# priority claims is left without one.
#
# A node is one of three kinds:
#   0  continue  - do nothing; fall through to the next priority (the default)
#   1  directive - assign this directive, if the vassal passes vanilla's gate
#   2  condition - test a condition and descend into the matching branch
#
# Nodes are heap-addressed within a priority: n1 is the root, n2/n3 are its
# branches, and n4..n7 are the grandchildren. n4..n7 are leaves and can only be
# "continue" or "directive". That depth cap is deliberate: the script language
# forbids a scripted effect from calling itself, so the walk cannot recurse and
# each level is instead written out as its own effect below.
#
# Rules live in variables on the player: leo_mvd_r<P>_n<N>_{kind,dir,cond,thresh}
# for priorities 1..6. leo_mvd_rule_count is how many of them are in use.
#
# Two things script cannot do shape the rest of this file: a variable name
# cannot be composed at runtime, and a condition cannot be chosen at runtime.
# So node access is macro-expanded with literal names ($NODE$), while the
# per-directive and per-condition chains - which would otherwise be duplicated
# into every node - are written once and driven by a value staged into a
# temporary scope.

### Staging.

# Current scope: vassal. Stages $NODE$'s directive for leo_mvd_try_assign_effect.
leo_mvd_stage_dir_effect = {
	save_temporary_scope_value_as = {
		name = leo_mvd_dir_id
		value = scope:leo_mvd_liege.var:leo_mvd_$NODE$_dir
	}
}

# Current scope: vassal. Stages $NODE$'s condition for leo_mvd_cond_holds_trigger.
leo_mvd_stage_cond_effect = {
	save_temporary_scope_value_as = {
		name = leo_mvd_cond_id
		value = scope:leo_mvd_liege.var:leo_mvd_$NODE$_cond
	}
	save_temporary_scope_value_as = {
		name = leo_mvd_thresh
		value = scope:leo_mvd_liege.var:leo_mvd_$NODE$_thresh
	}
}

### Assignment.

# Current scope: vassal. Requires scope:leo_mvd_dir_id (staged above).
# Assigns the staged directive if the vassal passes vanilla's gate for it, and
# does nothing at all otherwise - so an ineligible vassal simply falls through
# to the next priority, exactly as vanilla hides an unavailable send_option.
#
# The leo_mvd_hit guard is what stops later priorities from overwriting an
# earlier one's assignment; leo_mvd_assign_effect sets it.
leo_mvd_try_assign_effect = {
	if = {
		limit = { NOT = { has_variable = leo_mvd_hit } }

		if = {
			limit = {
				scope:leo_mvd_dir_id = 1
				leo_mvd_gate_convert_faith_trigger = yes
			}
			leo_mvd_assign_effect = { DIRECTIVE = convert_faith CODE = 1 }
		}
		else_if = {
			limit = {
				scope:leo_mvd_dir_id = 2
				leo_mvd_gate_convert_culture_trigger = yes
			}
			leo_mvd_assign_effect = { DIRECTIVE = convert_culture CODE = 2 }
		}
		else_if = {
			limit = {
				scope:leo_mvd_dir_id = 3
				leo_mvd_gate_improve_development_trigger = yes
			}
			leo_mvd_assign_effect = { DIRECTIVE = improve_development CODE = 3 }
		}
		else_if = {
			limit = {
				scope:leo_mvd_dir_id = 4
				leo_mvd_gate_train_commanders_trigger = yes
			}
			leo_mvd_assign_effect = { DIRECTIVE = train_commanders CODE = 4 }
		}
		else_if = {
			limit = {
				scope:leo_mvd_dir_id = 5
				leo_mvd_gate_build_maa_trigger = yes
			}
			leo_mvd_assign_effect = { DIRECTIVE = build_maa CODE = 5 }
		}
		else_if = {
			limit = {
				scope:leo_mvd_dir_id = 6
				leo_mvd_gate_improve_cultural_acceptance_trigger = yes
			}
			leo_mvd_assign_effect = { DIRECTIVE = improve_cultural_acceptance CODE = 6 }
		}
		else_if = {
			limit = {
				scope:leo_mvd_dir_id = 7
				leo_mvd_gate_building_focus_trigger = yes
			}
			leo_mvd_assign_effect = { DIRECTIVE = building_focus_fortification CODE = 7 }
		}
		else_if = {
			limit = {
				scope:leo_mvd_dir_id = 8
				leo_mvd_gate_building_focus_trigger = yes
			}
			leo_mvd_assign_effect = { DIRECTIVE = building_focus_military CODE = 8 }
		}
		else_if = {
			limit = {
				scope:leo_mvd_dir_id = 9
				leo_mvd_gate_building_focus_trigger = yes
			}
			leo_mvd_assign_effect = { DIRECTIVE = building_focus_economy CODE = 9 }
		}
	}
}

### The walk.
#
# One effect per tree level. These are near-identical by necessity: a scripted
# effect cannot call itself, so the recursion is unrolled by hand instead.

# Current scope: vassal. A leaf: continue, or assign a directive.
leo_mvd_eval_leaf_effect = {
	if = {
		limit = {
			scope:leo_mvd_liege = { var:leo_mvd_$NODE$_kind = 1 }
		}
		leo_mvd_stage_dir_effect = { NODE = $NODE$ }
		leo_mvd_try_assign_effect = yes
	}
}

# Current scope: vassal. A mid node: its branches must be leaves.
leo_mvd_eval_mid_effect = {
	if = {
		limit = {
			scope:leo_mvd_liege = { var:leo_mvd_$NODE$_kind = 2 }
		}
		leo_mvd_stage_cond_effect = { NODE = $NODE$ }
		if = {
			limit = { leo_mvd_cond_holds_trigger = yes }
			leo_mvd_eval_leaf_effect = { NODE = $LEFT$ }
		}
		else = {
			leo_mvd_eval_leaf_effect = { NODE = $RIGHT$ }
		}
	}
	else_if = {
		limit = {
			scope:leo_mvd_liege = { var:leo_mvd_$NODE$_kind = 1 }
		}
		leo_mvd_stage_dir_effect = { NODE = $NODE$ }
		leo_mvd_try_assign_effect = yes
	}
}

# Current scope: vassal. A priority's root node; its branches may be conditions.
leo_mvd_eval_root_effect = {
	if = {
		limit = {
			scope:leo_mvd_liege = { var:leo_mvd_$NODE$_kind = 2 }
		}
		leo_mvd_stage_cond_effect = { NODE = $NODE$ }
		if = {
			limit = { leo_mvd_cond_holds_trigger = yes }
			leo_mvd_eval_mid_effect = { NODE = $LEFT$ LEFT = $LL$ RIGHT = $LR$ }
		}
		else = {
			leo_mvd_eval_mid_effect = { NODE = $RIGHT$ LEFT = $RL$ RIGHT = $RR$ }
		}
	}
	else_if = {
		limit = {
			scope:leo_mvd_liege = { var:leo_mvd_$NODE$_kind = 1 }
		}
		leo_mvd_stage_dir_effect = { NODE = $NODE$ }
		leo_mvd_try_assign_effect = yes
	}
}

# Current scope: vassal. Evaluates priority $P$ of the main rule set.
# Priorities beyond leo_mvd_rule_count are skipped, so the unused slots a rule
# set does not fill cost nothing.
leo_mvd_eval_priority_effect = {
	if = {
		limit = {
			scope:leo_mvd_liege = { var:leo_mvd_rule_count >= $N$ }
		}
		leo_mvd_eval_root_effect = {
			NODE = $P$_n1
			LEFT = $P$_n2
			RIGHT = $P$_n3
			LL = $P$_n4
			LR = $P$_n5
			RL = $P$_n6
			RR = $P$_n7
		}
	}
}

### Editing.
#
# The panel cannot hand script a number, so "set node N's directive to D" cannot
# be one call. Instead a radio click runs two scripted GUIs in order: the first
# points leo_mvd_edit_node at the node, the second stages a field and value and
# calls the dispatch below. That makes the cost of the editor one scripted GUI
# per node plus one per option, rather than one per node-and-option pair.
#
# leo_mvd_edit_node addresses a node as priority * 10 + node, so r1_n1 is 11 and
# r6_n7 is 67.

# Current scope: the player. Writes the staged field to $NODE$.
leo_mvd_write_node_effect = {
	if = {
		limit = { var:leo_mvd_stage_field = 1 }
		set_variable = { name = leo_mvd_$NODE$_kind value = var:leo_mvd_stage_value }
	}
	else_if = {
		limit = { var:leo_mvd_stage_field = 2 }
		set_variable = { name = leo_mvd_$NODE$_cond value = var:leo_mvd_stage_value }
	}
	else_if = {
		limit = { var:leo_mvd_stage_field = 3 }
		set_variable = { name = leo_mvd_$NODE$_thresh value = var:leo_mvd_stage_value }
	}
	else_if = {
		limit = { var:leo_mvd_stage_field = 4 }
		set_variable = { name = leo_mvd_$NODE$_dir value = var:leo_mvd_stage_value }
	}
}

# Current scope: the player. Point the cursor at $NODE$.
leo_mvd_focus_effect = {
	set_variable = { name = leo_mvd_edit_node value = $CODE$ }
}

# Current scope: the player. Stage field $FIELD$ = $VALUE$ and write it to the
# focused node.
#
# Changing anything makes the rules the player's own, whatever preset they came
# from. That is what lets a preset be a starting point: tweak one of its rules
# and you keep the rest of it, rather than building the whole waterfall again to
# change one directive. A built-in cannot be edited in place because it would no
# longer be the preset it still claimed to be.
leo_mvd_edit_effect = {
	set_variable = { name = leo_mvd_stage_field value = $FIELD$ }
	set_variable = { name = leo_mvd_stage_value value = $VALUE$ }
	leo_mvd_write_field_effect = yes
	leo_mvd_after_edit_effect = yes
}

### Rule authoring.
#
# Used to build the rule sets below, and by the panel's editor.

# Current scope: the player. Reset one node to "continue to next priority".
leo_mvd_clear_node_effect = {
	set_variable = { name = leo_mvd_$NODE$_kind value = 0 }
	set_variable = { name = leo_mvd_$NODE$_dir value = 0 }
	set_variable = { name = leo_mvd_$NODE$_cond value = 0 }
	set_variable = { name = leo_mvd_$NODE$_thresh value = 0 }
}

# Current scope: the player. Reset every node of priority $P$.
leo_mvd_clear_priority_effect = {
	leo_mvd_clear_node_effect = { NODE = $P$_n1 }
	leo_mvd_clear_node_effect = { NODE = $P$_n2 }
	leo_mvd_clear_node_effect = { NODE = $P$_n3 }
	leo_mvd_clear_node_effect = { NODE = $P$_n4 }
	leo_mvd_clear_node_effect = { NODE = $P$_n5 }
	leo_mvd_clear_node_effect = { NODE = $P$_n6 }
	leo_mvd_clear_node_effect = { NODE = $P$_n7 }
}

# Current scope: the player. Reset the whole main rule set to empty. Also the
# one place that guarantees every rule variable exists, which the panel relies
# on: it reads them straight from GUI, where an undefined variable has no

leo_mvd_clear_rules_effect = {

	leo_mvd_clear_priority_effect = { P = r2 }
	leo_mvd_clear_priority_effect = { P = r3 }

	leo_mvd_clear_priority_effect = { P = r6 }
	set_variable = { name = leo_mvd_rule_count value = 0 }
}

# Current scope: the player. Make $NODE$ assign directive $DIR$.

# Current scope: the player. Make $NODE$ test condition $COND$ against $THRESH$

	set_variable = { name = leo_mvd_$NODE$_cond value = $COND$ }

# Current scope: the player. A priority the player has yet to fill in: one node,
# ready to assign a directive, with no directive chosen. An unset directive
# matches nothing, so it stays inert until it is picked.

	leo_mvd_clear_priority_effect = { P = $P$ }
	leo_mvd_set_node_dir_effect = { NODE = $P$_n1 DIR = 0 }
}

### The Custom slot.
#
# There is one live rule set, and selecting a preset overwrites it - so the
# player's own rules are kept aside in a parallel set while a built-in is
# loaded, and handed back when they return to Custom. Without this, trying a
# preset would cost you your own waterfall.

# Current scope: the player. Set the live rules aside as the player's own.
leo_mvd_backup_custom_effect = {
	leo_mvd_copy_priority_between_effect = { FROM = r1 TO = cr1 }
	leo_mvd_copy_priority_between_effect = { FROM = r2 TO = cr2 }
	leo_mvd_copy_priority_between_effect = { FROM = r3 TO = cr3 }
	leo_mvd_copy_priority_between_effect = { FROM = r4 TO = cr4 }
	leo_mvd_copy_priority_between_effect = { FROM = r5 TO = cr5 }
	leo_mvd_copy_priority_between_effect = { FROM = r6 TO = cr6 }
	set_variable = { name = leo_mvd_crule_count value = var:leo_mvd_rule_count }
	set_variable = leo_mvd_has_custom
}

# Current scope: the player. Put the player's own rules back.
# Only ever called when leo_mvd_has_custom says there are some.
leo_mvd_restore_custom_effect = {
	leo_mvd_copy_priority_between_effect = { FROM = cr1 TO = r1 }
	leo_mvd_copy_priority_between_effect = { FROM = cr2 TO = r2 }
	leo_mvd_copy_priority_between_effect = { FROM = cr3 TO = r3 }
	leo_mvd_copy_priority_between_effect = { FROM = cr4 TO = r4 }
	leo_mvd_copy_priority_between_effect = { FROM = cr5 TO = r5 }
	leo_mvd_copy_priority_between_effect = { FROM = cr6 TO = r6 }
	set_variable = { name = leo_mvd_rule_count value = var:leo_mvd_crule_count }
}

### Built-in rule sets.
#
# Built-ins are rule data, not hardcoded logic: each one loads a rule set that
# the engine then evaluates by the same path as a hand-built one. So there is a
# single implementation of the waterfall, and "Convert and Develop" reproducing
# the mod's original behavior doubles as the engine's regression test.
#
# Selecting one overwrites the rule set. leo_mvd_preset records which is loaded:
# 0 none, 1-4 built-in, 5 the player's own rules.

# Current scope: the player.
# No rules at all, and the default for a player who has not chosen yet. The mod
# assigns nothing until a preset is picked; the waterfall still runs when
# automation is on, but with no priorities it only clears directives the mod
# assigned earlier, leaving manual ones alone.
leo_mvd_preset_0_effect = {
	leo_mvd_clear_rules_effect = yes
	set_variable = { name = leo_mvd_preset value = 0 }
}

# Current scope: the player.
# Steer vassals onto your faith, then your culture, then grow them: strong
# vassals develop their land, weaker ones build up their military.
leo_mvd_preset_1_effect = {
	leo_mvd_clear_rules_effect = yes
	set_variable = { name = leo_mvd_rule_count value = 4 }

	# 1: Faith is Yours -> Convert Faith. Vanilla's gate only asks whether a
	# vassal holds an off-faith county, not whose faith they hold, so without
	# this condition an off-faith vassal would be told to spread their own.
	leo_mvd_set_node_cond_effect = { NODE = r1_n1 COND = 1 THRESH = 0 }
	leo_mvd_set_node_dir_effect = { NODE = r1_n2 DIR = 1 }

	# 2: Culture is Yours -> Promote Culture (the same reasoning)

# Current scope: the player. The player's own rules.
#
# Picking Custom from a preset starts blank - one priority, waiting for a
# directive - rather than carrying that preset over. Inheriting on selection
# sounds helpful but isn't: someone who picks Custom outright wants their own
# waterfall, and would have to dismantle four priorities of someone else's
# first. Starting from a preset is a different gesture, and it already works:
# change one of its rules and leo_mvd_edit_effect makes the whole thing yours.
#
# Doing nothing when Custom is already selected matters for exactly that reason
# - otherwise picking Custom over rules that are already yours would wipe them.
leo_mvd_preset_5_effect = {
	if = {
		limit = { NOT = { var:leo_mvd_preset = 5 } }
		if = {
			# Rules of their own, set aside when they last left Custom.
			limit = { has_variable = leo_mvd_has_custom }
			leo_mvd_restore_custom_effect = yes
		}
		else = {
			leo_mvd_clear_rules_effect = yes
			set_variable = { name = leo_mvd_rule_count value = 1 }
			leo_mvd_init_priority_effect = { P = r1 }
		}
	}
	set_variable = { name = leo_mvd_preset value = 5 }
}

# Current scope: the player. Load preset $N$ and re-apply if automation is on.
#
# Leaving Custom sets the player's rules aside first, so that trying a preset
# does not cost them their own waterfall.
leo_mvd_select_preset_effect = {
	if = {
		limit = { var:leo_mvd_preset = 5 }
		leo_mvd_backup_custom_effect = yes
	}
	leo_mvd_preset_$N$_effect = yes
	if = {
		limit = { has_variable = leo_mvd_auto }
		leo_mvd_run_waterfall_effect = yes
	}
}

### Adding and removing priorities.

# Current scope: the player. Everything an edit implies: the rules become the
# player's own, and automation picks the change up at once.
leo_mvd_after_edit_effect = {
	set_variable = { name = leo_mvd_preset value = 5 }
	if = {
		limit = { has_variable = leo_mvd_auto }
		leo_mvd_run_waterfall_effect = yes
	}
}

# Current scope: the player. Append an empty priority to the waterfall.
leo_mvd_add_priority_effect = {
	if = {
		limit = { var:leo_mvd_rule_count < 6 }
		change_variable = { name = leo_mvd_rule_count add = 1 }
		# Which slot that just became. A variable name cannot be built from a
		# value, so the new priority has to be named outright.
		if = {
			limit = { var:leo_mvd_rule_count = 2 }
			leo_mvd_init_priority_effect = { P = r2 }
		}
		else_if = {
			limit = { var:leo_mvd_rule_count = 3 }
			leo_mvd_init_priority_effect = { P = r3 }
		}
		else_if = {
			limit = { var:leo_mvd_rule_count = 4 }
			leo_mvd_init_priority_effect = { P = r4 }
		}
		else_if = {
			limit = { var:leo_mvd_rule_count = 5 }
			leo_mvd_init_priority_effect = { P = r5 }
		}
		else_if = {
			limit = { var:leo_mvd_rule_count = 6 }
			leo_mvd_init_priority_effect = { P = r6 }
		}
		leo_mvd_after_edit_effect = yes
	}
}

# Current scope: the player. Shared tail of removing a priority: the ones below
# it have already moved up, so the last slot is now a duplicate.
leo_mvd_finish_remove_effect = {
	leo_mvd_clear_priority_effect = { P = r6 }
	change_variable = { name = leo_mvd_rule_count add = -1 }
	leo_mvd_after_edit_effect = yes
}

# Current scope: the heir. Carries the whole rule set over from root, including
# the Custom slot - an heir who inherits a preset should still find their
# predecessor's own rules waiting under Custom.
leo_mvd_copy_rules_effect = {
	set_variable = { name = leo_mvd_preset value = root.var:leo_mvd_preset }
	set_variable = { name = leo_mvd_rule_count value = root.var:leo_mvd_rule_count }
	leo_mvd_copy_priority_effect = { P = r1 }
	leo_mvd_copy_priority_effect = { P = r2 }
	leo_mvd_copy_priority_effect = { P = r3 }
	leo_mvd_copy_priority_effect = { P = r4 }
	leo_mvd_copy_priority_effect = { P = r5 }
	leo_mvd_copy_priority_effect = { P = r6 }
	if = {
		limit = { root = { has_variable = leo_mvd_has_custom } }
		set_variable = leo_mvd_has_custom
		set_variable = { name = leo_mvd_crule_count value = root.var:leo_mvd_crule_count }
		leo_mvd_copy_priority_effect = { P = cr1 }
		leo_mvd_copy_priority_effect = { P = cr2 }
		leo_mvd_copy_priority_effect = { P = cr3 }
		leo_mvd_copy_priority_effect = { P = cr4 }

	}

# Leo VI's Mass Vassal Directives - scripted effects
#
# Directive assignment matches vanilla give_vassal_directive_interaction
# on_accept exactly: remove_vassal_directives (vanilla scripted effect,
# 00_realm_effects.txt) followed by add_character_flag.
#
# Directive codes (stored in leo_mvd_managed and in each rule node's dir):
# 1 convert_faith, 2 convert_culture, 3 improve_development,
# 4 train_commanders, 5 build_maa, 6 improve_cultural_acceptance,
# 7 building_focus_fortification, 8 building_focus_military,
# 9 building_focus_economy. 0 means "continue to the next priority".
#
# Which directive a vassal gets is decided by the rule engine in
# leo_mvd_rules.txt; this file is the machinery around it.

# Current scope: vassal. Assign a directive and mark it as mod-managed.
# leo_mvd_hit records that this vassal has been claimed, so that the priorities
# after the one that assigned leave the vassal alone.
leo_mvd_assign_effect = {
	if = {
		limit = {
			NOT = { has_character_flag = vassal_directive_$DIRECTIVE$ }
		}
		remove_vassal_directives = yes
		add_character_flag = vassal_directive_$DIRECTIVE$
	}
	set_variable = {
		name = leo_mvd_managed
		value = $CODE$
	}
	set_variable = leo_mvd_hit
}

# Current scope: vassal. Clear the directive only if the mod still owns it.
# If the flag no longer matches the stored code, the player re-assigned it
# manually via the vanilla interaction - drop ownership, keep the directive.
leo_mvd_clear_stale_effect = {
	if = {
		limit = { has_variable = leo_mvd_managed }
		if = {
			limit = { leo_mvd_managed_matches_trigger = yes }
			remove_vassal_directives = yes
		}
		remove_variable = leo_mvd_managed
	}
}

# Current scope: vassal. Requires scope:leo_mvd_liege (the player).
# Run the liege's rule set over this vassal: each priority in turn until one
# assigns a directive. A vassal no priority claims keeps nothing the mod gave
# them earlier.
leo_mvd_evaluate_vassal_effect = {
	leo_mvd_eval_priority_effect = { P = r1 N = 1 }
	leo_mvd_eval_priority_effect = { P = r2 N = 2 }
	leo_mvd_eval_priority_effect = { P = r3 N = 3 }
	leo_mvd_eval_priority_effect = { P = r4 N = 4 }
	leo_mvd_eval_priority_effect = { P = r5 N = 5 }
	leo_mvd_eval_priority_effect = { P = r6 N = 6 }

	if = {
		limit = { has_variable = leo_mvd_hit }
		remove_variable = leo_mvd_hit
	}
	else = {
		leo_mvd_clear_stale_effect = yes
	}
}

# Current scope: the player. Run the waterfall across all eligible vassals.
leo_mvd_run_waterfall_effect = {
	save_scope_as = leo_mvd_liege
	if = {
		limit = { government_allows = administrative }
		every_vassal_or_below = {
			limit = {
				leo_mvd_base_eligible_trigger = { LIEGE = scope:leo_mvd_liege }
			}
			leo_mvd_evaluate_vassal_effect = yes
		}
	}
	else = {
		every_vassal = {
			limit = {
				leo_mvd_base_eligible_trigger = { LIEGE = scope:leo_mvd_liege }
			}
			leo_mvd_evaluate_vassal_effect = yes
		}
	}
}

# Current scope: the player. Escape hatch: remove every mod-assigned directive
# and exemption, then disable the mod. Manually assigned directives are kept.
leo_mvd_remove_all_effect = {
	every_vassal_or_below = {
		limit = {
			OR = {
				has_variable = leo_mvd_managed
				has_variable = leo_mvd_exempt
			}
		}
		leo_mvd_clear_stale_effect = yes
		remove_variable = leo_mvd_exempt
	}
	remove_variable = leo_mvd_auto
}

# Current scope: the player. Idempotent config defaults.
#
# The default is deliberately no rules and no preset selected: the mod must
# never assign a directive the player did not ask for, so it does nothing at all
# until a preset is picked.
#
# Clearing the rule set writes every rule variable, so this one call is also
# what guarantees they all exist - which the panel depends on, since it reads
# them directly from GUI rather than through a scripted GUI.
leo_mvd_init_defaults_effect = {
	if = {
		limit = { NOT = { has_variable = leo_mvd_preset } }
		leo_mvd_preset_0_effect = yes
	}
	# Editor state: which node the panel is writing to, and the pending write.
	if = {
		limit = { NOT = { has_variable = leo_mvd_edit_node } }
		set_variable = { name = leo_mvd_edit_node value = 0 }
		set_variable = { name = leo_mvd_stage_field value = 0 }
		set_variable = { name = leo_mvd_stage_value value = 0 }
	}
}

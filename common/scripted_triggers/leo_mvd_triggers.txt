# Leo VI's Mass Vassal Directives - scripted triggers
#
# Eligibility must match vanilla exactly. Wherever possible we call the vanilla
# scripted triggers directly (00_vassal_stance_triggers.txt). The remaining
# conditions are verbatim mirrors of give_vassal_directive_interaction
# (00_vassal_interactions.txt:3428-3453 and its send_option blocks) because the
# interaction's inline is_shown/is_valid blocks cannot be invoked from script.
# Re-diff those mirrored lines against vanilla after each game patch.

# Current scope: vassal candidate. $LIEGE$ = the liege issuing directives.
#
# Verbatim mirror of give_vassal_directive_interaction's is_shown
# (00_vassal_interactions.txt:3428-3444) - the vassals vanilla will even offer a
# directive for. Re-diff this against vanilla after each game patch; it is the
# one block both the rules and the exempt interaction depend on, so a drift here
# is a drift everywhere.
leo_mvd_directive_shown_trigger = {
	trigger_if = {
		limit = {
			$LIEGE$ = { government_allows = administrative }
		}
		is_vassal_or_below_of = $LIEGE$
	}
	trigger_else = {
		is_vassal_of = $LIEGE$
	}
	is_landed = yes
	NOT = { government_has_flag = government_is_herder }
	highest_held_title_tier >= tier_county
	is_ai = yes
}

# Current scope: vassal candidate. $LIEGE$ = the liege issuing directives.
# Everything vanilla asks before a directive may be given, plus the mod's own
# exemption. Mirrors the interaction's is_shown + is_valid +
# is_valid_showing_failures_only.
leo_mvd_base_eligible_trigger = {
	leo_mvd_directive_shown_trigger = { LIEGE = $LIEGE$ }
	# is_valid_showing_failures_only mirror (:3450-3453); the follows/valid
	# sub-triggers are both included in the vanilla composite below
	is_physically_able = yes
	# is_valid mirror - the canonical vanilla eligibility composite
	vassal_is_valid_and_follows_directive_trigger = { VASSAL = this LIEGE = $LIEGE$ }
	# mod-side: skip vassals the player exempted from automation
	NOT = { has_variable = leo_mvd_exempt }
}

### Per-directive gates.
#
# One trigger per directive, each a verbatim mirror of that directive's
# send_option block in give_vassal_directive_interaction. These are pure
# vanilla: whose faith or culture a vassal should be steered toward is a
# targeting choice, expressed as a rule condition below, and is never baked in
# here.
#
# A vassal who fails the gate for the directive a rule picked is left alone and
# falls through to the next priority, which is what vanilla does when a
# send_option is unavailable.

# Current scope: vassal. improve_development (:3455)
leo_mvd_gate_improve_development_trigger = {
	government_allows = administrative
	highest_held_title_tier >= tier_duchy
}

# Current scope: vassal. train_commanders (:3469)
leo_mvd_gate_train_commanders_trigger = {
	government_allows = administrative
}

# Current scope: vassal. build_maa (:3480)
leo_mvd_gate_build_maa_trigger = {
	government_allows = administrative
}

# Current scope: vassal. convert_faith (:3491)
leo_mvd_gate_convert_faith_trigger = {
	save_temporary_scope_as = leo_mvd_vassal_tmp
	NOT = { government_has_flag = government_is_nomadic }
	any_sub_realm_county = {
		faith != scope:leo_mvd_vassal_tmp.faith
	}
}

# Current scope: vassal. convert_culture (:3510)
leo_mvd_gate_convert_culture_trigger = {
	save_temporary_scope_as = leo_mvd_vassal_tmp
	NOT = { government_has_flag = government_is_nomadic }
	any_sub_realm_county = {
		culture != scope:leo_mvd_vassal_tmp.culture
	}
}

# Current scope: vassal. improve_cultural_acceptance (:3529)
leo_mvd_gate_improve_cultural_acceptance_trigger = {
	save_temporary_scope_as = leo_mvd_vassal_tmp
	NOT = { government_has_flag = government_is_nomadic }
	highest_held_title_tier >= tier_duchy
	any_sub_realm_county = {
		culture != scope:leo_mvd_vassal_tmp.culture
	}
}

# Current scope: vassal. building_focus_fortification (:3549),
# building_focus_military (:3560) and building_focus_economy (:3571) all share
# the same single vanilla condition.
leo_mvd_gate_building_focus_trigger = {
	NOT = { government_has_flag = government_is_nomadic }
}

### Nomad directives.
#
# The four directives only nomads can be given. Every directive above requires
# either an administrative government or a non-nomadic one, so these and those
# divide the eligible vassals exactly between them - which is why the nomad
# rules are their own waterfall rather than a special case inside the main one.
#
# Herders are not part of this: vanilla's is_shown excludes them from directives
# altogether (:3440), and leo_mvd_base_eligible_trigger mirrors that.

# Current scope: vassal. manage_fertility (:3582). Vanilla asks nothing further.
leo_mvd_gate_manage_fertility_trigger = {
	government_has_flag = government_is_nomadic
}

# Current scope: vassal. explore_cultures (:3592)
leo_mvd_gate_explore_cultures_trigger = {
	save_temporary_scope_as = leo_mvd_vassal_tmp
	government_has_flag = government_is_nomadic
	any_sub_realm_county = {
		NOT = { culture = scope:leo_mvd_vassal_tmp.culture }
	}
}

# Current scope: vassal. raid_innovation_intent (:3610)
#
# Vanilla also refuses this to a vassal who already has it - alone among the
# thirteen. That is a menu nicety: it greys out an option that would do nothing.
# Mirrored literally it would mean the mod could never leave the directive in
# place - the vassal holding it would fail their own rule, fall through, and
# have it taken off them a month after it was given. What the rules can reach is
# unchanged, since a vassal with this directive is one vanilla would allow; only
# the pointless re-offer is dropped.
leo_mvd_gate_raid_innovation_intent_trigger = {
	government_has_flag = government_is_nomadic
}

# Current scope: vassal. raid_herd_intent (:3628). Vanilla asks nothing further.
leo_mvd_gate_raid_herd_intent_trigger = {
	government_has_flag = government_is_nomadic
}

### Rule conditions.
#
# The closed set of conditions a rule can branch on, evaluated in the vassal's
# scope with the liege as scope:leo_mvd_liege. Script cannot dispatch on a
# condition chosen at runtime, so this is a single value-driven chain rather
# than one trigger per node.
#
# Every numeric condition reads "is at least" (>=) so the true branch is always
# the high side; no condition gets its own comparison direction.
#
# Requires scope:leo_mvd_cond_id (the condition's code) and scope:leo_mvd_thresh
# (its threshold, 0 for the boolean conditions), both staged by
# leo_mvd_stage_cond_effect.
leo_mvd_cond_holds_trigger = {
	save_temporary_scope_as = leo_mvd_vassal_tmp

	# 1 - Faith is Yours
	trigger_if = {
		limit = { scope:leo_mvd_cond_id = 1 }
		faith = scope:leo_mvd_liege.faith
	}
	# 2 - Culture is Yours
	trigger_else_if = {
		limit = { scope:leo_mvd_cond_id = 2 }
		culture = scope:leo_mvd_liege.culture
	}
	# 3 - Holds Counties of Another Faith
	trigger_else_if = {
		limit = { scope:leo_mvd_cond_id = 3 }
		any_sub_realm_county = {
			faith != scope:leo_mvd_vassal_tmp.faith
		}
	}
	# 4 - Holds Counties of Another Culture
	trigger_else_if = {
		limit = { scope:leo_mvd_cond_id = 4 }
		any_sub_realm_county = {
			culture != scope:leo_mvd_vassal_tmp.culture
		}
	}
	# 5 - Same Dynasty as You
	trigger_else_if = {
		limit = { scope:leo_mvd_cond_id = 5 }
		has_dynasty = yes
		scope:leo_mvd_liege = { has_dynasty = yes }
		dynasty = scope:leo_mvd_liege.dynasty
	}
	# 6 - Administrative Government
	trigger_else_if = {
		limit = { scope:leo_mvd_cond_id = 6 }
		government_allows = administrative
	}
	# 7 - Is a Powerful Vassal
	trigger_else_if = {
		limit = { scope:leo_mvd_cond_id = 7 }
		is_powerful_vassal_of = scope:leo_mvd_liege
	}
	# 8 - Is on Your Council
	trigger_else_if = {
		limit = { scope:leo_mvd_cond_id = 8 }
		is_councillor_of = scope:leo_mvd_liege
	}
	# 9 - Military Strength is at least. The chosen value is the duchy baseline;
	# leo_mvd_effective_threshold scales it by the vassal's tier.
	trigger_else_if = {
		limit = { scope:leo_mvd_cond_id = 9 }
		max_military_strength >= leo_mvd_effective_threshold
	}
	# 10 - Title Tier is at least. Compared against the named tier constants so
	# their numeric mapping never has to be assumed.
	trigger_else_if = {
		limit = { scope:leo_mvd_cond_id = 10 }
		trigger_if = {
			limit = { scope:leo_mvd_thresh = 2 }
			highest_held_title_tier >= tier_county
		}
		trigger_else_if = {
			limit = { scope:leo_mvd_thresh = 3 }
			highest_held_title_tier >= tier_duchy
		}
		trigger_else_if = {
			limit = { scope:leo_mvd_thresh = 4 }
			highest_held_title_tier >= tier_kingdom
		}
		trigger_else = {
			highest_held_title_tier >= tier_empire
		}
	}
	# 11 - Capital Development is at least
	trigger_else_if = {
		limit = { scope:leo_mvd_cond_id = 11 }
		exists = capital_county
		capital_county = {
			development_level >= scope:leo_mvd_thresh
		}
	}
	# 12 - Opinion of You is at least
	trigger_else_if = {
		limit = { scope:leo_mvd_cond_id = 12 }
		opinion = {
			target = scope:leo_mvd_liege
			value >= scope:leo_mvd_thresh
		}
	}
	# 13 - Counties Held is at least
	trigger_else_if = {
		limit = { scope:leo_mvd_cond_id = 13 }
		any_held_county = {
			count >= scope:leo_mvd_thresh
		}
	}
	# 14 - Cultural Acceptance with You is at least
	trigger_else_if = {
		limit = { scope:leo_mvd_cond_id = 14 }
		culture = {
			cultural_acceptance = {
				target = scope:leo_mvd_liege.culture
				value >= scope:leo_mvd_thresh
			}
		}
	}
	# 19 - Same House as You. Narrower than condition 5: this is the one house,
	# not the dynasty and its cadet branches. Both sides are guarded because a
	# lowborn has no house and an unguarded scope comparison writes to the log.
	trigger_else_if = {
		limit = { scope:leo_mvd_cond_id = 19 }
		exists = house
		exists = scope:leo_mvd_liege.house
		house = scope:leo_mvd_liege.house
	}
	# 20 - Governor Theme is. The only condition whose picked value selects a
	# value rather than setting a floor: themes are unordered, so this reads as
	# equality where every other measured condition reads "is at least".
	# vassal_contract_has_flag is a flag query on the contract, not a scope
	# change, so it simply answers no for a vassal who is not a governor.
	trigger_else_if = {
		limit = { scope:leo_mvd_cond_id = 20 }
		is_governor = yes
		trigger_if = {
			limit = { scope:leo_mvd_thresh = 1 }
			vassal_contract_has_flag = admin_theme_balanced
		}
		trigger_else_if = {
			limit = { scope:leo_mvd_thresh = 2 }
			vassal_contract_has_flag = admin_theme_civilian
		}
		trigger_else_if = {
			limit = { scope:leo_mvd_thresh = 3 }
			vassal_contract_has_flag = admin_theme_military
		}
		trigger_else_if = {
			limit = { scope:leo_mvd_thresh = 4 }
			vassal_contract_has_flag = admin_theme_frontier
		}
		trigger_else_if = {
			limit = { scope:leo_mvd_thresh = 5 }
			vassal_contract_has_flag = admin_theme_imperial
		}
		trigger_else = {
			vassal_contract_has_flag = admin_theme_naval
		}
	}
	# 21 - Average Development is at least. Averaged over the counties the
	# vassal holds personally: directives are not passed down to sub-vassals,
	# so those are the counties a directive could actually improve. The average
	# is staged as a scope first because a computed value cannot stand on the
	# left of a comparison by itself. A vassal holding no county (a baron) is
	# answered no outright, which also keeps the average's divisor above zero.
	trigger_else_if = {
		limit = { scope:leo_mvd_cond_id = 21 }
		any_held_county = { always = yes }
		save_temporary_scope_value_as = {
			name = leo_mvd_avg_dev
			value = leo_mvd_avg_held_development
		}
		scope:leo_mvd_avg_dev >= scope:leo_mvd_thresh
	}
	# @AGOT:conditions@
	# No condition picked yet: the branch is inert rather than silently true.
	trigger_else = {
		always = no
	}
}
